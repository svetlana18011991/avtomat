    <script>
        const baseUrl = 'https://raw.githubusercontent.com/svetlana18011991/avtomat_igrovoy/main/';
        const gameContainer = document.getElementById('game-container');
        const coin = document.getElementById('coin');
        const clawContainer = document.getElementById('claw-container');
        const clawImg = document.getElementById('claw-img');
        const clawCord = document.getElementById('claw-cord');
        const progressArea = document.getElementById('progress-area');

        const btnLeft = document.getElementById('btn-left');
        const btnRight = document.getElementById('btn-right');
        const btnAction = document.getElementById('btn-action');
        const controls = [btnLeft, btnRight, btnAction];

        let gameState = 'waiting_coin';
        let collectedToysCount = 0;
        let selectedToyElement = null;
        let grabbedToyOffsetY = 0;

        const clawPositionsX = [39.38, 45.90, 52.02];
        const clawStartTop = 20.14;
        const clawWidth = 7.46;
        const clawHeight = 20.27;
        let currentCol = 1;

        const tasks = [
            { q: '5 + 4 = ?', options: ['8', '9', '10'], a: '9' },
            { q: '12 - 5 = ?', options: ['6', '7', '8'], a: '7' },
            { q: '3 × 3 = ?', options: ['6', '9', '12'], a: '9' },
            { q: '16 ÷ 4 = ?', options: ['3', '4', '5'], a: '4' },
            { q: '8 + 7 = ?', options: ['14', '15', '16'], a: '15' },
            { q: '10 - 8 = ?', options: ['1', '2', '3'], a: '2' },
            { q: '2 × 6 = ?', options: ['10', '12', '14'], a: '12' },
            { q: '9 ÷ 3 = ?', options: ['2', '3', '4'], a: '3' },
            { q: '10 + 10 = ?', options: ['10', '20', '30'], a: '20' }
        ];

        for (let i = 0; i < tasks.length; i++) {
            let slot = document.createElement('div');
            slot.className = 'collected-slot';
            slot.id = 'slot-' + i;
            progressArea.appendChild(slot);
        }

        function updateClawPosition(targetLeft, targetTop) {
            clawContainer.style.left = targetLeft + '%';
            clawContainer.style.top = targetTop + '%';

            clawCord.style.left = (targetLeft + (clawWidth / 2)) + '%';
            let cordHeight = targetTop - 12 + 2;
            if (cordHeight < 0) cordHeight = 0;
            clawCord.style.height = cordHeight + '%';
        }

        function getPercentPositionFromRect(element) {
            const containerRect = gameContainer.getBoundingClientRect();
            const rect = element.getBoundingClientRect();

            return {
                left: ((rect.left - containerRect.left) / containerRect.width) * 100,
                top: ((rect.top - containerRect.top) / containerRect.height) * 100,
                width: (rect.width / containerRect.width) * 100,
                height: (rect.height / containerRect.height) * 100
            };
        }

        setTimeout(() => {
            updateClawPosition(clawPositionsX[1], clawStartTop);
        }, 100);

        coin.addEventListener('click', () => {
            if (gameState !== 'waiting_coin') return;

            document.getElementById('sound-coin').play();
            gameState = 'playing';
            coin.classList.remove('blinking');
            coin.style.opacity = '0';
            controls.forEach(c => c.classList.add('active'));
        });

        btnLeft.addEventListener('click', () => {
            if (gameState === 'playing' && currentCol > 0) {
                currentCol--;
                updateClawPosition(clawPositionsX[currentCol], clawStartTop);
            }
        });

        btnRight.addEventListener('click', () => {
            if (gameState === 'playing' && currentCol < 2) {
                currentCol++;
                updateClawPosition(clawPositionsX[currentCol], clawStartTop);
            }
        });

        btnAction.addEventListener('click', () => {
            if (gameState !== 'playing') return;
            gameState = 'claw_moving';

            const toysInCol = Array.from(document.querySelectorAll(`.toy[data-col="${currentCol}"]`))
                .filter(toy => !toy.classList.contains('collected'));

            if (toysInCol.length === 0) {
                document.getElementById('error-modal').classList.add('active');
                gameState = 'playing'; // Исправлено, чтобы игра не зависала
                return;
            }

            const randomIndex = Math.floor(Math.random() * toysInCol.length);
            selectedToyElement = toysInCol[randomIndex];

            const toyPos = getPercentPositionFromRect(selectedToyElement);
            const clawCenterX = clawPositionsX[currentCol] + clawWidth / 2;

            // 1. ОПУСКАЕМ КЛЕШНЮ ДО САМОЙ ИГРУШКИ!
            const targetClawTop = toyPos.top - (clawHeight * 0.5);
            grabbedToyOffsetY = clawHeight * 0.5;

            updateClawPosition(clawPositionsX[currentCol], targetClawTop);

            setTimeout(() => {
                clawImg.src = baseUrl + '10.png';

                setTimeout(() => {
                    clawImg.src = baseUrl + 'клешня.png';

                    // Игрушка встает по центру клешни, но по высоте остается на своем месте (не подпрыгивает!)
                    selectedToyElement.style.left = (clawCenterX - toyPos.width / 2) + '%';
                    selectedToyElement.style.top = (targetClawTop + grabbedToyOffsetY) + '%';
                    selectedToyElement.style.zIndex = '25';
                    selectedToyElement.style.pointerEvents = 'none';

                    setTimeout(() => {
                        // 2. Клешня и игрушка едут наверх СИНХРОННО (без прыжков)
                        updateClawPosition(clawPositionsX[currentCol], clawStartTop);
                        selectedToyElement.style.top = (clawStartTop + grabbedToyOffsetY) + '%';

                        setTimeout(() => {
                            showTask();
                        }, 1000);

                    }, 250);
                }, 400);
            }, 1000);
        });

        document.getElementById('btn-close-error').addEventListener('click', () => {
            document.getElementById('error-modal').classList.remove('active');
            gameState = 'playing';
        });

        function showTask() {
            gameState = 'task_active';
            const task = tasks[collectedToysCount];
            document.getElementById('task-text').textContent = task.q;

            const optionsDiv = document.getElementById('task-options');
            optionsDiv.innerHTML = '';

            task.options.forEach(opt => {
                const btn = document.createElement('div');
                btn.className = 'task-option';
                btn.textContent = opt;
                btn.onclick = () => checkAnswer(opt, task.a, btn);
                optionsDiv.appendChild(btn);
            });

            document.getElementById('task-modal').classList.add('active');
        }

        function checkAnswer(selected, correct, btnElement) {
            if (selected === correct) {
                btnElement.classList.add('correct');
                document.getElementById('sound-success').play();

                setTimeout(() => {
                    document.getElementById('task-modal').classList.remove('active');
                    dropToy();
                }, 1000);
            } else {
                btnElement.classList.add('incorrect');
                document.getElementById('sound-error').play();

                setTimeout(() => btnElement.classList.remove('incorrect'), 500);
            }
        }

        function dropToy() {
            selectedToyElement.style.left = '45.16%';
            selectedToyElement.style.top = '71.76%';
            selectedToyElement.style.width = '5.58%';
            selectedToyElement.style.height = '11.22%';
            selectedToyElement.style.zIndex = '25';
            selectedToyElement.style.pointerEvents = 'auto';
            selectedToyElement.classList.add('collected');

            setTimeout(() => {
                selectedToyElement.style.animation = 'blink 2s infinite';
                selectedToyElement.onclick = () => {
                    moveToProgress(selectedToyElement);
                };
            }, 1000);
        }

        function moveToProgress(toy) {
            toy.style.animation = 'none';
            toy.classList.add('toy-collected-anim');
            document.getElementById('sound-coin').play();

            setTimeout(() => {
                toy.style.display = 'none';

                const slot = document.getElementById('slot-' + collectedToysCount);
                slot.innerHTML = `<img src="${toy.src}">`;
                collectedToysCount++;

                if (collectedToysCount < tasks.length) {
                    coin.style.opacity = '1';
                    coin.classList.add('blinking');
                    controls.forEach(c => c.classList.remove('active'));
                    gameState = 'waiting_coin';
                } else {
                    setTimeout(() => alert('Победа! Все игрушки собраны!'), 500);
                }
            }, 500);
        }
    </script>
