  function packPayload(p) {
      const out = {};
      if (p.title && p.title !== "Игровой автомат") out.ti = p.title;
      if (p.bgDataUrl) out.bg = p.bgDataUrl;
      if (p.useLatex) out.ul = 1;
      if (p.timer && p.timer.enabled) { out.tm = 1; out.ts = p.timer.seconds; }
      if (p.instruction) {
          out.i = {};
          if (p.instruction.text !== "Вставь монетку и управляй клешней.\nПоймал игрушку — реши задачу.\nВерно → игрушка твоя.\nНеверно → игрушка упадет обратно.\nУ тебя ровно 9 попыток!") out.i.t = p.instruction.text;
          if (Object.keys(out.i).length === 0) delete out.i;
      }
      if (p.feedback && p.feedback.text) out.f = p.feedback.text;
      out.q = (p.questions || []).map(q => {
          const mq = {}; mq.t = q.type === "text" ? 1 : 0;
          if (q.prompt) mq.p = q.prompt;
          if (mq.t === 0) { if (q.options) mq.o = q.options; if (q.correctIndices && (q.correctIndices.length > 1 || q.correctIndices[0] !== 0)) mq.c = q.correctIndices; } else { if (q.accepts) mq.a = q.accepts; }
          if (q.seconds) mq.s = q.seconds; if (q.image) mq.i = q.image; if (q.audio) mq.au = q.audio; if (q.font) mq.fn = q.font; if (q.fontSize && q.fontSize !== 32) mq.fs = q.fontSize;
          return mq;
      });
      return out;
  }

  function unpackPayload(out) {
      if (!out || typeof out !== 'object') return normalizeCfg({});
      if (out.questions) return normalizeCfg(out); 
      const p = {
          title: out.ti || "Игровой автомат", useLatex: !!out.ul, bgDataUrl: out.bg || "", timer: { enabled: !!out.tm, seconds: out.ts || 30 },
          instruction: { text: (out.i && out.i.t !== undefined) ? out.i.t : "Вставь монетку и управляй клешней.\nПоймал игрушку — реши задачу.\nВерно → игрушка твоя.\nНеверно → игрушка упадет обратно.\nУ тебя ровно 9 попыток!" },
          feedback: { text: out.f || "" },
          questions: (out.q || []).map(mq => {
              const q = {}; q.type = mq.t === 1 ? "text" : "choice"; q.prompt = mq.p || "";
              if (q.type === "choice") { q.options = mq.o || ["Да", "Нет"]; q.correctIndices = mq.c || [0]; } else { q.accepts = mq.a || ["Ответ"]; }
              q.seconds = mq.s || null; q.image = mq.i || ""; q.audio = mq.au || ""; q.font = mq.fn = q.font; q.fontSize = mq.fs || null; return q;
          })
      };
      return normalizeCfg(p);
  }

  function encodeGameData(payload) { return "~" + LZString.compressToEncodedURIComponent(JSON.stringify(packPayload(payload))); }
  function decodeGameData(str) {
    if (str.startsWith("~")) return unpackPayload(JSON.parse(LZString.decompressFromEncodedURIComponent(str.slice(1))));
    let s = String(str||"").replaceAll("-","+").replaceAll("_","/"); while(s.length % 4) s += "=";
    return unpackPayload(JSON.parse(decodeURIComponent(escape(atob(s))))); 
  }

  function validateQuestion(q){
    if(!q) return "Пустой вопрос"; if(!(q.prompt||"").trim()) return "Введите текст";
    if(q.type === "choice"){
      const opts = (q.options||[]).map(x=>String(x||"").trim()).filter(Boolean); if(opts.length < 2) return "Минимум 2 варианта"; q.options = opts;
      if(!q.correctIndices || q.correctIndices.length === 0) return "Отметьте верный ответ";
      q.correctIndices = q.correctIndices.filter(i => i >= 0 && i < q.options.length);
      if(q.correctIndices.length === 0) q.correctIndices = [0]; return "";
    }
    const acc = (q.accepts||[]).map(x=>String(x||"").trim()).filter(Boolean); if(acc.length === 0) return "Отметьте верный ответ"; q.accepts = acc; return "";
  }

  function buildPayload(previewWithDraft){
    let questions = Editor.questions.map(q=>{
      if(q.type === "choice"){ return { type:"choice", prompt:q.prompt||"", options:(q.options||[]).slice(), correctIndices:(q.correctIndices||[0]).slice(), seconds:(q.seconds??null), image:q.image||"", audio:q.audio||"", font:q.font||"", fontSize:q.fontSize||null }; }
      return { type:"text", prompt:q.prompt||"", accepts:(q.accepts||[]).slice(), seconds:(q.seconds??null), image:q.image||"", audio:q.audio||"", font:q.font||"", fontSize:q.fontSize||null };
    });
    if(previewWithDraft && Editor.draft){
      const d = deepClone(Editor.draft); questions = questions.slice();
      if(d.type === "choice"){ const mapped = { type:"choice", prompt:d.prompt||"", options:(d.options||[]).slice(), correctIndices:(d.correctIndices||[0]).slice(), seconds: (d.seconds==null || d.seconds==="") ? null : Number(d.seconds), image:d.image||"", audio:d.audio||"", font:d.font||"", fontSize:d.fontSize||null }; if(questions.length === 0) questions.push(mapped); else questions[0] = mapped; }

  }

  window.processImport = () => {
      const txt = document.getElementById('importText').value.trim();
      if (!txt) return closeImport();
      const mode = document.getElementById('importMode').value; const tplIdx = parseInt(document.getElementById('newQTemplate').value);
      let baseQ = { font: "", fontSize: 32, seconds: null, image: "", audio: "" };
      if (tplIdx >= 0 && Editor.questions[tplIdx]) { const tq = Editor.questions[tplIdx]; baseQ = { font: tq.font, fontSize: tq.fontSize, seconds: tq.seconds }; }
      let addedCount = 0;
      if (mode === 'line') {
          const lines = txt.split('\n');
          for(let line of lines) { if (Editor.questions.length >= 9) break; const trimmed = line.trim(); if (trimmed) { Editor.questions.push({ id: uid(), ...baseQ, type: 'text', prompt: trimmed, accepts: ['Ответ'] }); addedCount++; } }
      } else {
          const blocks = txt.split(/\n\s*\n/);
          for(let block of blocks) {
              if (Editor.questions.length >= 9) break;
              const lines = block.trim().split('\n'); let qText = []; let opts = []; let autoAnswer = '';
              const optRegex = /^(\d+|[а-яА-Яa-zA-Z])[\)\.]\s+(.+)$/;
              lines.forEach(line => {
                  const trimmed = line.trim(); const match = trimmed.match(optRegex);
                  if (match) { opts.push({ marker: match[1].toLowerCase(), t: match[2].trim() }); } 
                  else if (trimmed.toLowerCase().startsWith('ответ:') || trimmed.toLowerCase().startsWith('answer:')) { autoAnswer = trimmed.replace(/^(ответ:|answer:)\s*/i, '').trim().toLowerCase(); } 
                  else { qText.push(trimmed); }
              });
              if (opts.length >= 2) {
                  let correctIndices = [0];
                  if (autoAnswer) { const foundIdx = opts.findIndex(o => o.marker === autoAnswer || o.t.toLowerCase() === autoAnswer); if (foundIdx !== -1) correctIndices = [foundIdx]; }
                  Editor.questions.push({ id: uid(), ...baseQ, type: 'choice', prompt: qText.join('\n'), options: opts.map(o => o.t), correctIndices: correctIndices }); addedCount++;
              } else if (qText.length > 0) { Editor.questions.push({ id: uid(), ...baseQ, type: 'text', prompt: qText.join('\n'), accepts: autoAnswer ? [autoAnswer] : ['Ответ'] }); addedCount++; }
          }
      }
      closeImport(); renderList(); updatePreview(false);
      if (Editor.questions.length >= 9 && addedCount > 0) toast("Импорт завершен (достигнут лимит 9)");
