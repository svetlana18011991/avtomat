  function colorDistanceSq(a, b){
    const dr = a[0] - b[0], dg = a[1] - b[1], db = a[2] - b[2];
    return dr*dr + dg*dg + db*db;

  function removePlainBackgroundFromCanvas(canvas){
    const ctx = canvas.getContext("2d", { willReadFrequently: true });
    const w = canvas.width, h = canvas.height;
    if(!w || !h) throw new Error("empty-canvas");
    const img = ctx.getImageData(0, 0, w, h);
    const data = img.data;
    const idxOf = (x, y) => (y * w + x) * 4;
    const sample = (x, y) => { const i = idxOf(x, y); return [data[i], data[i+1], data[i+2], data[i+3]]; };
    const seeds = [[0,0], [w-1,0], [0,h-1], [w-1,h-1]];
    const bgColors = seeds.map(([x,y]) => sample(x,y));
    const tol = 58, tolSq = tol * tol;
    const visited = new Uint8Array(w * h);
    const queue = [];
    const isBg = (x, y) => {
      const i = idxOf(x, y);
      const alpha = data[i+3];
      if(alpha <= 12) return true;
      const p = [data[i], data[i+1], data[i+2], alpha];
      return bgColors.some(c => c[3] <= 12 || colorDistanceSq(p, c) <= tolSq);
    };
    for(const [x,y] of seeds){ const k = y*w + x; visited[k] = 1; queue.push([x,y]); }
    let head = 0, removed = 0;
    while(head < queue.length){
      const [x,y] = queue[head++];
      if(!isBg(x,y)) continue;
      const di = idxOf(x,y); data[di+3] = 0; removed++;
      const neigh = [[x+1,y], [x-1,y], [x,y+1], [x,y-1]];
      for(const [nx,ny] of neigh){
        if(nx < 0 || ny < 0 || nx >= w || ny >= h) continue;
        const k = ny*w + nx;
        if(visited[k]) continue;
        visited[k] = 1;
        queue.push([nx,ny]);
      }
    }
    const alphaCopy = new Uint8ClampedArray(w * h);
    for(let i=0, k=0; i<data.length; i+=4, k++) alphaCopy[k] = data[i+3];
    for(let y=1; y<h-1; y++){
      for(let x=1; x<w-1; x++){
        const k = y*w+x, i = k*4;
        if(alphaCopy[k] === 0) continue;
        let nearTransparent = false;
        for(let yy=-1; yy<=1; yy++){
          for(let xx=-1; xx<=1; xx++){
            if(alphaCopy[(y+yy)*w + (x+xx)] === 0) nearTransparent = true;
          }
        }
        if(nearTransparent && isBg(x, y)) data[i+3] = Math.min(data[i+3], 35);
      }
    }
    ctx.putImageData(img, 0, 0);
    return removed;
  }
